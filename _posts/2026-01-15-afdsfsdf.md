using UnityEngine.SceneManagement;


  public enum SceneName
  {
      FirstScene,
      TodayScene,
  }

  // SceneManager.LoadScene(); -> 오버로딩 기능 (5개 정도 있는 듯)

  SceneManager.LoadScene((int)SceneName.TodayScene);  // 열거형(enum) 으로 씬 로드
  SceneManager.LoadScene("20260115_Scene"); // 씬 명칭으로 로드, 그러나 명칭은 오타날 수도 있음
