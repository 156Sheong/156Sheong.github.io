```csharp
// 아래 순서에 따라 프로세스 실행될 때, 특정 위치(location) 프로세스 언제 실행되는지(answer) 보기
//  1. 처음에 모든 프로세스 큐에 넣기
//  2. 앞에서부터 순차적으로 프로세스 꺼내보며, 우선도(클수록 높음)에 따라 프로세스 1번씩 실행
public class Solution 
{
    public int solution(int[] priorities, int location)
    {
        int answer = 0; // 시도 횟수
        Queue<ProcessClass> queProcess = new Queue<ProcessClass>(); // 초기 위치, 우선순위 합한 클래스
        // Queue<(int, int)> queProcess = new Queue<(int, int)>();  // 튜플로 시도했다 클래스로 변경

        
        // 큐 넣기
        for(int i = 0; i < priorities.Length; i++)
        {
            queProcess.Enqueue(new ProcessClass(i, priorities[i]));
            
        }
            
        
        // break 되기 전까지는 계속 반복
        while(true)
        {
            ProcessClass processClass = null;   // 큐에 담을 클래스
            int currentMaxNumber = 0;   // 가장 높은 숫자 찾기
            
            
            // 가장 높은 숫자 판단
            foreach(ProcessClass num in queProcess)
            {
                if (num.Priority > currentMaxNumber) 
                {
                    currentMaxNumber = num.Priority;
                }
            }
                    
            
            // 가장 앞의 큐 꺼내기
            processClass = queProcess.Dequeue();
            
            
            // 가장 높은 우선 순위라면 시도 횟수 늘리고 대상인지 판단
            if(processClass.Priority == currentMaxNumber)
            {
                // 시도 횟수 1로 시작하기 위해 ++
                answer++;
                
                // 대상 아이템이면 반복 끝내기
                if(processClass.Index == location)
                {
                    break;
                }
                
                // Dequeue 라서 max 숫자라면 대상이 아니어도 자동으로 제거됨
            }  
            // 대상 아이템 아니면 큐에 다시 넣기
            else
            {
                queProcess.Enqueue(processClass);
            }
            
            
        }
        
   
        return answer;
    }
    
}



public class ProcessClass
{
    public int Index { get; private set; }  // 로케이션
    public int Priority { get; private set; }   // 중요도
    
    public ProcessClass(int id, int priority)
    {
        Index = id;
        Priority = priority;
    }
}
```

```csharp

public class Solution 
{
    public int solution(int[] priorities, int location)
    {
        int answer = 0;
        
        // 높은 순서대로 우선순위 정렬한 기준 배열 생성
        int[] sortedPriorities = (int[])priorities.Clone(); // int[] 타입으로 캐스팅(int[]) 한 새 배열 생성(.Clone())
        Array.Sort(sortedPriorities);
        Array.Reverse(sortedPriorities); 

        int cursor = 0;      // 배열을 순회할 커서
        int targetIndex = 0; // 현재 가장 높은 중요도

        
        while (true)
        {
            // 현재 위치가 최고 우선 순위인지 확인
            if (priorities[cursor] == sortedPriorities[targetIndex])
            {
                // 최고 우선 순위가 맞으면 실행 횟수 +1
                answer++; 

                // 찾던 로케이션이라면 반복 종료하기
                if (cursor == location)
                {
                    break;
                }

                // 타겟이 아니라면 다음 우선순위 찾아 인덱스 +1
                targetIndex++;
            }

            // 큐에서 빼고 넣는 작업 없이 커서 만 1칸 옆으로 이동
            // 배열 끝에 도달하면 나머지 연산으로 0번 인덱스로 되돌리기
            cursor = (cursor + 1) % priorities.Length;
        }

        return answer;
    }
    
}
```
