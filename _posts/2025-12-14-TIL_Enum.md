## B. 열거형의 정의와 기초 이해
- 열거형 형식은 기본적인 정수 숫자 형식을 사용하여 정의된 값 형식
- 이를 통해, 숫자에 벌명을 붙여 알아보기 쉽게 만들어줄 수 있음



## C. 열거형의 선언 방법 및 사용 방법
- `enum`을 사용하고 `열거형 멤버의 이름`을 지정하여 열거형을 사용
- `enum`은 기본적으로 0으로 시작, 그러나 번호를 바꿀 수 있음
- 열거형 사용 예시
  - 다양한 종족과 국가가 존재하는 세계관에서 각 종족과 세력을 열거형으로 구분
  ```csharp
  // 열거형 선언
  enum Factions // 같은 종족인데 다른 문화권이 추가될 수 있기에, 명칭은 {종족} + {정치체 / 지리 등 문화}로 지음
  {
    HumanKingdom = 1, // 1 = 왕국 인가
    HumanHighland = 2,  // 2 = 고지 인간
    ElfEmpire,  // 3 = 제국 엘프
    ElfForest,  // 4 = 숲 엘프
    OrcSteppe,  // 5 = 스텝 오크
    OrcDessert,  // 6 = 사막 오크
  }

  // 열거형 사용

  static void Main()
  {
  // enum 변수 선언 및 값 할당 가능
  Factions yourFaction;
  yourFaction = Factions.HumanKingdom;

  // 콘솔 입력으로 열거형 값 할당 가능
  Factions enemyFaction;
  Enum.TryParse(Console.ReadLine(), out enemyFaction);    // 직접 입력 시, 

      // 열거형을 사용하기 위한 switch 문
      switch (yourFaction)
      {
          case Factions.HumanKingdom:
              Console.WriteLine("You are Human Kingdom.");
              break;
          case Factions.HumanHighland:
              Console.WriteLine("You are Human Highland.");
              break;
          case Factions.ElfEmpire:
              Console.WriteLine("You are Elf Empire.");
              break;
          case Factions.ElfForest:
              Console.WriteLine("You are Elf Forest.");
              break;
          case Factions.OrcSteppe:
              Console.WriteLine("You are Orc Steppe.");
              break;
          case Factions.OrcDessert:
              Console.WriteLine("You are Orc Dessert.");
              break;
          default:
              Console.WriteLine("Unknown faction."); // 발생할 일 없어야 하는 기본 값
              break;
      }

      // switch 패턴 일치 식
      string factiongString = yourFaction switch
      {
          Factions.HumanKingdom => "You are Human Kingdom.",
          Factions.HumanHighland => "You are Human Highland.",
          Factions.ElfEmpire => "You are Elf Empire.",
          Factions.ElfForest => "You are Elf Forest.",
          Factions.OrcSteppe => "You are Orc Steppe.",
          Factions.OrcDessert => "You are Orc Dessert.",
          _ => "Unknown faction." // 발생할 일 없어야 하는 기본 값
      };

      Console.WriteLine(factiongString);
  }
  ```
